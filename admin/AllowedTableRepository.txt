package com.mas.sap.prak.admin.repository;

import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.util.UUID;

public class AllowedTableRepository {

    private final DataSource dataSource;

    public AllowedTableRepository(
            DataSource dataSource
    ) {

        this.dataSource = dataSource;
    }

    public void insertTable(
            UUID tableId,
            UUID apiId,
            String tableName,
            String aliasName
    ) throws Exception {

        String sql =
                """
                insert into api_allowed_table
                (
                    table_id,
                    api_id,
                    table_name,
                    alias_name
                )
                values
                (?, ?, ?, ?)
                """;

        try (
                Connection connection =
                        dataSource.getConnection();

                PreparedStatement ps =
                        connection.prepareStatement(sql)
        ) {

            ps.setObject(1, tableId);
            ps.setObject(2, apiId);
            ps.setString(3, tableName);
            ps.setString(4, aliasName);

            ps.executeUpdate();
        }
    }
}
