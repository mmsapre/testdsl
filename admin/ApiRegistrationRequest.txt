package com.mas.sap.prak.admin.dto;

import java.util.List;

public record ApiRegistrationRequest(
        String domain,
        String apiName,
        String apiPath,
        String method,
        String datasourceKey,
        List<TableRequest> tables
) {}

public record TableRequest(
        String tableName,
        String aliasName,
        List<ColumnRequest> columns
) {}

public record ColumnRequest(
        String columnName,
        String columnAlias,
        String dataType,
        boolean selectable,
        boolean filterable,
        boolean sortable
) {}
