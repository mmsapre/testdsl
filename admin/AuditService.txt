package com.mas.sap.prak.audit;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.mas.sap.prak.security.RequestContext;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Service;

import java.util.Map;
import java.util.UUID;

@Service
public class AuditService {

    private final JdbcTemplate jdbcTemplate;
    private final ObjectMapper objectMapper = new ObjectMapper();

    public AuditService(JdbcTemplate jdbcTemplate) {
        this.jdbcTemplate = jdbcTemplate;
    }

    public void audit(
            String apiPath,
            RequestContext ctx,
            Object request,
            Object response,
            int statusCode,
            long durationMs
    ) {
        try {
            jdbcTemplate.update("""
                insert into api_audit_log
                (audit_id, api_path, tenant_id, consumer_id,
                 request_body, response_body, status_code, duration_ms)
                values (?, ?, ?, ?, ?::jsonb, ?::jsonb, ?, ?)
                """,
                    UUID.randomUUID(),
                    apiPath,
                    ctx.tenantId(),
                    ctx.consumerId(),
                    objectMapper.writeValueAsString(request),
                    objectMapper.writeValueAsString(response),
                    statusCode,
                    durationMs
            );
        } catch (Exception ignored) {
            // audit failure should not break runtime request
        }
    }
}
