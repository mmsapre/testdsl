package com.mas.sap.prak.datasource;

public class RequestDataSourceContext {

    private static final ThreadLocal<String> DATASOURCE =
            new ThreadLocal<>();

    public static void set(
            String datasourceKey
    ) {

        DATASOURCE.set(datasourceKey);
    }

    public static String get() {

        return DATASOURCE.get();
    }

    public static void clear() {

        DATASOURCE.remove();
    }
}
