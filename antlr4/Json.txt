grammar Json;
json : value EOF ;
value : object | array | STRING | NUMBER | TRUE | FALSE | NULL ;
object : '{' pair (',' pair)* '}' ;
pair : STRING ':' value ;
array : '[' value (',' value)* ']' ;
STRING : '"' (~["\\] | '\\' .)* '"' ;
NUMBER : '-'? [0-9]+ ('.' [0-9]+)? ;
TRUE:'true'; FALSE:'false'; NULL:'null';
WS : [ \t\r\n]+ -> skip ;
